import re
text = input()
total_kcal = 0

regex = r"([\#\|])([A-Za-z ]+)\1(\d{2}/\d{2}/\d{2})\1(\d+)\1"
result = re.findall(regex, text)

for i in result:
    kcal = int(i[3])
    total_kcal += kcal
days = total_kcal // 2000
print(f"You have food to last you for: {days} days!")

for item in result:
    name = item[1]
    date = item[2]
    kcal = item[3]
    print(f"Item: {name}, Best before: {date}, Nutrition: {kcal}")
