initial_pieces = int(input())

collection = {}

for i in range(initial_pieces):
    command = input()
    piece, composer, key = command.split('|')
    if piece not in collection:
        collection[piece] = [composer, key]


while True:
    command = input()
    if command == 'Stop':
        break
    command = command.split('|')
    command_type = command[0]
    if command_type == 'Add':
        piece = command[1]
        composer = command[2]
        key = command[3]
        if piece in collection:
            print(f"{piece} is already in the collection!")
        else:
            collection[piece] = [composer, key]
            print(f"{piece} by {composer} in {key} added to the collection!")


    elif command_type == 'Remove':
        piece = command[1]
        if piece in collection:
            collection.pop(piece)
            print(f"Successfully removed {piece}!")
        else:
            print(f"Invalid operation! {piece} does not exist in the collection.")

    elif command_type == 'ChangeKey':
        piece = command[1]
        new_key = command[2]

        if piece in collection:
            collection[piece][1] = new_key
            print(f"Changed the key of {piece} to {new_key}!")
        else:
            print(f"Invalid operation! {piece} does not exist in the collection.")

for piece in collection:
    print(f"{piece} -> Composer: {collection[piece][0]}, Key: {collection[piece][1]}")
